证书基本检查

public static Result<Void> checkBundleId(String bundleId, byte[] certFile, String certPassword) {
        Assert.hasText(bundleId, "参数bundleId不能为空");
        if (StringUtils.isEmpty(certPassword)) {
            return Result.<Void>builder().code(StatusCode.ERROR).msg("未设置证书密码").build();
        }
        if (certFile == null || certFile.length == 0) {
            log.warn("参数certFile为空");
            return Result.<Void>builder().code(StatusCode.ERROR).msg("参数certFile为空").build();
        }
        try {
            KeyStore p12 = KeyStore.getInstance("pkcs12");
            ByteArrayInputStream inputStream = new ByteArrayInputStream(certFile);
            p12.load(inputStream, certPassword.toCharArray());
            Enumeration<String> e = p12.aliases();
            String certBundleId;
            while (e.hasMoreElements()) {
                String alias = e.nextElement();
                X509Certificate c = (X509Certificate) p12.getCertificate(alias);
                //过期时间
                Date notAfter = c.getNotAfter();
                if (notAfter.before(new Date())) {
                    String str = DateUtils.parseDate2LocalDate(notAfter, Constants.DATE_FORMAT_YMD_CN_HMS);
                    return Result.<Void>builder().code(StatusCode.ERROR).msg("证书已过期 到期时间为" + str).build();
                }
                Principal subject = c.getSubjectDN();
                String name = subject.getName();
                int index = name.indexOf("UID=");
                if (index != -1) {
                    certBundleId = name.substring(index + 4);
                    if (certBundleId.equals(bundleId)) {
                        return sslCheck(p12, certPassword);
                    } else {
                        return Result.<Void>builder().code(StatusCode.ERROR).msg("校验失败 bundleId不匹配 应用bundleId=" + bundleId + " 证书bundleId=" + certBundleId).build();
                    }
                }
            }
            return Result.<Void>builder().code(StatusCode.ERROR).msg("校验失败 获取证书bundleId失败").build();
        } catch (KeyStoreException e) {
            log.error("获取pkcs12失败", e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("获取pkcs12失败").build();
        } catch (CertificateException e) {
            log.error("bundleId={}证书异常", bundleId, e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("bundleId=" + bundleId + " 证书异常").build();
        } catch (NoSuchAlgorithmException e) {
            log.error("加密算法不存在", e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("加密算法不存在").build();
        } catch (IOException e) {
            if (e.getMessage().contains("keystore password was incorrect")) {
                return Result.<Void>builder().code(StatusCode.ERROR).msg("证书密码错误").build();
            }
            log.error("bundleId={}证书文件异常", bundleId, e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("bundleId=" + bundleId + "证书文件异常").build();
        }

    }

证书吊销检查

public static Result<Void> sslCheck(@NonNull KeyStore p12, @NonNull String certPassword) {
        try {
            KeyManagerFactory keyMgrFactory = KeyManagerFactory.getInstance("SunX509");
            keyMgrFactory.init(p12, certPassword.toCharArray());
            SSLContext sslContext = SSLContext.getInstance("TLS");
            sslContext.init(keyMgrFactory.getKeyManagers(), null, null);
            SSLSocketFactory sslSocketFactory = sslContext.getSocketFactory();
             // 默认正式  2197 instead of port 443
            SSLSocket sslSocket = (SSLSocket)sslSocketFactory.createSocket(ApnsClientBuilder.PRODUCTION_APNS_HOST, 443);
            String[] cipherSuites = sslSocket.getSupportedCipherSuites();
            sslSocket.setEnabledCipherSuites(cipherSuites);
            sslSocket.startHandshake(); // 握手
            sslSocket.close();
        } catch (KeyStoreException e) {
            log.error("ssl KeyStore", e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("(KeyStore").build();
        } catch (IOException e) {
            log.error("ssl 检查证书失败", e);
            if (e.getMessage().contains("certificate_revoked")) {
                return Result.<Void>builder().code(StatusCode.ERROR).msg("证书被吊销").build();
            }
            return Result.<Void>builder().msg("证书异常").code(StatusCode.ERROR).build();
        } catch (NoSuchAlgorithmException e) {
            log.error("ssl 算法不可用", e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("(NoSuchAlgorithm 算法不可用").build();
        } catch (UnrecoverableKeyException e) {
            log.error("ssl UnrecoverableKey", e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("UnrecoverableKey").build();
        } catch (KeyManagementException e) {
            log.error("ssl KeyManagement", e);
            return Result.<Void>builder().code(StatusCode.ERROR).msg("KeyManagement").build();
        }
        return Result.<Void>builder().code(StatusCode.SUCCESS).build();
    }
