java线程池取消指定任务

单个任务取消

最近遇到一个业务场景，需要取消正在执行的任务，不想用quartz就自己参考ThreadPoolExecutor workers 实现了一个

自定义work

@Slf4j
public  class PushWorker extends AbstractQueuedSynchronizer implements Runnable {
    private static final long serialVersionUID = 6138294804551838833L;
    private final Thread thread;
    private final Runnable firstTask;
    private final String msgId; //对应msgId

    public String getMsgId() {
        return msgId;
    }
    public PushWorker(Runnable firstTask, String msgId, ThreadFactory threadFactory) {
        this.setState(-1);
        this.firstTask = firstTask;
        this.thread = threadFactory.newThread(this);
        this.msgId = msgId;
    }
    @Override
    public void run() {
        this.firstTask.run();
    }

    public void start() {
        this.thread.start();
    }
    protected boolean isHeldExclusively() {
        return this.getState() != 0;
    }

    protected boolean tryAcquire(int unused) {
        if (this.compareAndSetState(0, 1)) {
            this.setExclusiveOwnerThread(Thread.currentThread());
            return true;
        } else {
            return false;
        }
    }

    protected boolean tryRelease(int unused) {
        this.setExclusiveOwnerThread(null);
        this.setState(0);
        return true;
    }

   public void interruptIfStarted() {
        log.info("{} pushWorker停止msgId={}",this.thread.getName(),msgId);
        Thread t;
        if ((t = this.thread) != null && !t.isInterrupted()) {
            try {
                t.checkAccess();
                t.interrupt();
            } catch (SecurityException var3) {
                log.error("pushWorker停止异常",var3);
            }
        }
    }
}

开始任务

//线程池
@Autowired
private ThreadPoolTaskExecutor taskExecutor;
//推送work
private CopyOnWriteArrayList<PushWorker> workers = new CopyOnWriteArrayList<>();
//推送
ThreadFactory threadFactory = taskExecutor.getThreadPoolExecutor().getThreadFactory();
PushWorker pushWorker = new PushWorker(() -> {
   pushService.pushMessage(pushTaskDetail);
}, (String) msgId, threadFactory
);
workers.add(pushWorker);
pushWorker.start();

取消任务

String msgId="xxx";
ThreadPoolExecutor threadPoolExecutor = taskExecutor.getThreadPoolExecutor();
BlockingQueue<Runnable> queue = threadPoolExecutor.getQueue();
for (PushWorker worker : workers) {
   if (worker.getMsgId().equals(msgId)) {
     if (!queue.remove(worker)) {//缓存队列没有
       worker.interruptIfStarted();//停止线程
     }
     workers.remove(worker);
   }
}
